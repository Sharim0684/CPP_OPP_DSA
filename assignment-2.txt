Q5- Write a program to input a three-digit number and display the sum of the digits

  #include <stdio.h>
int main()
{
    int n, sum = 0, m;
    printf("Enter 3-digit number:");
    scanf("%d", &n);
    while (n > 0)
    {
        m = n % 10;
        sum = sum + m;
        n = n / 10;
    }
    printf("Sum of 3-digit is=%d", sum);
    return 0;
}


---->
     output = Enter 3-digit number:564
              Sum of 3-digit is=15
